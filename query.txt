SELECT datetime, valor, clave_inst
FROM base_ana
WHERE clave_inst = 
	Any(SELECT distinct(key) FROM base_instalation WHERE type = '2')
	AND LOWER(punto) LIKE LOWER('Nivel')
	AND valor >= 0
ORDER BY 
	datetime ASC,
	clave_inst ASC
